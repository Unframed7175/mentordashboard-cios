      setBpvImportError(null);
      const path = await open({ multiple: false, filters: [{ name: 'Excel', extensions: ['xls','xlsx'] }] });
      if (!path || typeof path !== 'string') return;
      try {
        const bytes = await readFile(path);
        const data = parseBpvExcel(bytes.buffer as ArrayBuffer);
        if (Object.keys(data).length === 0) {
          setBpvImportError('Onbekend BPV-bestandsformaat. Probeer een ander bestand.');
          return;
        }
        await saveBpvData(data);
      } catch (e) {
        console.warn('[SettingsPage] BPV import failed:', e);
        setBpvImportError('Onbekend BPV-bestandsformaat. Probeer een ander bestand.');
      }
