  const [editingNaamId, setEditingNaamId] = useState<string | null>(null);

  const [confirmingReset, setConfirmingReset] = useState(false);

Mount effect: useEffect(()=>{ Promise.all([getDeelgebiedenConfig(), getLeerlijnenMapping()])

Handler contracts (all instant-apply per Phase 17 pattern — update state THEN await save):
  handleNaamCommit(id, newLabel):
    if newLabel.trim() === '' → revert to current cfg label (no save)
    else: const next = dgConfig.map(c => c.id===id ? { ...c, label: newLabel.trim() } : c);
          setDgConfig(next); await saveDeelgebiedenConfig(next);
    setEditingNaamId(null);
  handleLeerlijnChange(id, newLeerlijn):  // newLeerlijn ∈ 'lesgeven'|'organiseren'|'profHandelen'
    const next = { ...leerlijnenMapping, [id]: newLeerlijn };
    setLeerlijnenMapping(next); await saveLeerlijnenMapping(next);
  handleActiefToggle(id, active):
    const next = dgConfig.map(c => c.id===id ? { ...c, active } : c);
    setDgConfig(next); await saveDeelgebiedenConfig(next);
  handleResetConfirm():
    await resetDeelgebiedenConfig();
    await resetLeerlijnenMapping();
    // re-fetch
    const [cfg, map] = await Promise.all([getDeelgebiedenConfig(), getLeerlijnenMapping()]);
    setDgConfig(cfg); setLeerlijnenMapping(map);
    setConfirmingReset(false);
