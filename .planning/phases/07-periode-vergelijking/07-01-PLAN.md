**Part A — klassen.js: Update `getActiveStudents()`** (lines 167-174)

Replace the current `getActiveStudents` with a version that returns one record per `leerlingId` (the most recent by alphabetical `periode` sort per D-07/D-11):

```javascript
window.getActiveStudents = function() {
  if (!window.klassenState.activeKlasId) return [];
  var klas = window.klassenState.klassen[window.klassenState.activeKlasId];
  if (!klas) return [];
  // Deduplicate to one (most recent) record per leerlingId for grid display (D-11)
  var sorted = klas.students.slice().sort(function(a, b) {
    return (b.periode || '').localeCompare(a.periode || '');
  });
  var seen = {};
  var result = [];
  for (var i = 0; i < sorted.length; i++) {
    if (!seen[sorted[i].leerlingId]) {
      seen[sorted[i].leerlingId] = true;
      result.push(sorted[i]);
    }
  }
  return result;
};
```

Add a convenience function `getAllRecordsForStudent` below `getActiveStudents` for use by the detail view (Plan 02 will consume this):

```javascript
/**
 * Get all StudentRecords for a given leerlingId in the active class,
 * sorted oldest-first by periode string (alphabetical).
 * Used by buildDetailDeelgebieden for multi-period comparison.
 * @param {string} leerlingId
 * @returns {StudentRecord[]}
 */
window.getAllRecordsForStudent = function(leerlingId) {
  if (!window.klassenState.activeKlasId) return [];
  var klas = window.klassenState.klassen[window.klassenState.activeKlasId];
  if (!klas) return [];
  return klas.students.filter(function(s) {
    return s.leerlingId === leerlingId;
  }).sort(function(a, b) {
    return (a.periode || '').localeCompare(b.periode || '');
  });
};
```

```javascript
function showDetail(leerlingId) {
    // Get the most recent record for this leerlingId (D-07: prognose uses most-recent)
    var allRecords = window.appState.students.filter(function(s) { return s.leerlingId === leerlingId; });
    if (allRecords.length === 0) return;
    allRecords.sort(function(a, b) { return (b.periode || '').localeCompare(a.periode || ''); });
    var student = allRecords[0]; // most recent
    // Pitfall 4 mitigation: if most-recent record lacks verzuim, inherit from any record that has it
    if (!student.verzuim) {
      for (var i = 1; i < allRecords.length; i++) {
        if (allRecords[i].verzuim) { student = Object.assign({}, student, { verzuim: allRecords[i].verzuim }); break; }
      }
    }
    detailStudentId = leerlingId;
    if (!detailStudentList.includes(leerlingId)) {
      detailStudentList = window.getActiveStudents().map(function(s) { return s.leerlingId; });
    }
    detailView.innerHTML = buildDetailHTML(student);
    showView('detail');
    wireDetailEvents();
  }
```

Key changes:
- Uses `filter` + `sort` to find the most-recent-period record (instead of `.find` which returns first match)
- Pitfall 4 mitigation: if the most-recent record has no `verzuim`, copies it from an older record that does (so `buildDetailVerzuim` always sees verzuim data when it exists for any period)
- Fallback `detailStudentList` uses `getActiveStudents()` (deduplicated) instead of raw `appState.students` — fixes Pitfall 5 (duplicate leerlingIds in prev/next nav)
