name: CI

on:
  push:
    branches:
      - main
  workflow_dispatch: {}

(No permissions block — omit entirely; GITHUB_TOKEN is auto-provided at default read level.)

jobs:
  build:
    strategy:
      fail-fast: false
      matrix:
        include:
          - platform: macos-latest
            args: '--target aarch64-apple-darwin'
            rust-targets: 'aarch64-apple-darwin'
          - platform: windows-latest
            args: '--bundles nsis --target x86_64-pc-windows-msvc'
            rust-targets: ''

    runs-on: ${{ matrix.platform }}

    steps:
      - uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: lts/*

      - name: Install Rust stable
        uses: dtolnay/rust-toolchain@stable
        with:
          targets: ${{ matrix.rust-targets }}

      - name: Install frontend dependencies
        run: npm ci

      - name: Build
        uses: tauri-apps/tauri-action@v0
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
          APPLE_SIGNING_IDENTITY: '-'
        with:
          args: ${{ matrix.args }}

CRITICAL CONSTRAINTS:
- The tauri-apps/tauri-action with: block must contain ONLY "args: ${{ matrix.args }}".
  Do NOT include tagName, releaseName, releaseBody, releaseDraft, or prerelease — their presence
  causes the action to create a GitHub Release on every main push (Pitfall 1 from RESEARCH.md).
- APPLE_SIGNING_IDENTITY: '-' must remain in the env block — omitting it causes macOS binary
  signing to fail (Pitfall 2 from RESEARCH.md).
- Keep only 2 matrix entries: macos-latest aarch64 + windows-latest x86_64.
  Do NOT add x86_64-apple-darwin (macOS Intel) — STATE.md specifies Apple Silicon only for CI.
- workflow_dispatch: {} is kept to allow manual re-runs without a push.
  </action>
