**saveKlassen (vervang regels 81–96):**
  export async function saveKlassen(): Promise<boolean>
  Body:
    const payload = { klassen: klassenState.klassen, activeKlasId: klassenState.activeKlasId, savedAt: new Date().toISOString() };
    const plaintext = JSON.stringify(payload);
    const ciphertext = await invoke<string>('encrypt_klassen', { plaintext });
    await store.set('klassen', ciphertext);
    await store.save();   // VERPLICHT — set() is alleen in-memory
    return true;
  Catch: console.error('[klassen.ts] saveKlassen mislukt:', e); return false;

**loadKlassen (vervang regels 98–132):**
  export async function loadKlassen(): Promise<boolean>
  Body:
    const ciphertext = await store.get<string>('klassen');
    if (ciphertext) {
      const plaintext = await invoke<string>('decrypt_klassen', { ciphertext });
      const data = JSON.parse(plaintext);
      if (data && data.klassen) { klassenState.klassen = data.klassen; klassenState.activeKlasId = data.activeKlasId || null; _restoreBridge(); return Object.keys(klassenState.klassen).length > 0; }
    }
    return _migrateLocalStorageToStore();
  Catch: console.error + showStorageError('Sleutel niet beschikbaar — neem contact op met beheerder'); return false;

**_migrateLocalStorageToStore (vervang _migrateV1ToKlassen, regels 134–165):**
  async function _migrateLocalStorageToStore(): Promise<boolean>
  Logica (per PATTERNS.md §Migration pattern):
    1. Lees rawV2 = localStorage.getItem(KLASSEN_KEY_V2), rawV1 = localStorage.getItem(KLASSEN_KEY_V1)
    2. raw = rawV2 || rawV1; if (!raw) return false
    3. Parse oldData; reconstructeer klassenState op basis van formaat (v2: oldData.klassen; v1: maak 'klas_' + Date.now().toString(36) aan met oldData.students)
    4. _restoreBridge()
    5. const saved = await saveKlassen(); if (!saved) { reset klassenState; return false; }
    6. ALLEEN na confirmed save: if (rawV2) localStorage.removeItem(KLASSEN_KEY_V2); if (rawV1) localStorage.removeItem(KLASSEN_KEY_V1)
    7. console.log('[klassen.ts] Migratie localStorage → plugin-store geslaagd'); return true
